package utils;

import org.junit.runner.RunWith;
import org.junit.runners.Parameterized;
import org.junit.Test;
import static org.junit.jupiter.api.Assertions.assertEquals;

import java.util.Arrays;
import java.util.Collection;

@RunWith(Parameterized.class)
public class Tester {
	private int luku;
	private String tulos;

	@Parameterized.Parameters(name="{0}s => {1}")
	public static Collection<Object[]> testiTapaukset() {
		return Arrays.asList(new Object[][] { 
			{ -1, "-1" },  
			{ 0, "00:00:00" }, 
			{ 1, "00:00:01" }, 
			{ 10, "00:00:10" }, 
			{ 59, "00:00:59" },
			{ 60, "00:01:00" }, 
			{ 600, "00:10:00" }, 
			{ 3599, "00:59:59" },
			{ 3600, "01:00:00" }, 
			{ 18000, "05:00:00" },
			{ 36000, "10:00:00" }, 
			{ 86399, "23:59:59" },
			{ 86400, "1d 00:00:00" }, 
			{ 604799, "6d 23:59:59" }, 
			{ 604800, "1w 0d 00:00:00" }, 
			{ 31535999, "52w 0d 23:59:59" }, 
			{ 31536000, "1y 0w 0d 00:00:00" }, 
		});
	}

	public Tester(int luku, String tulos) {
		this.luku = luku;
		this.tulos = tulos;
	}

	@Test
	public void testTime() {
		assertEquals(this.tulos, TimeUtils.secToTime(luku));
	}
}
